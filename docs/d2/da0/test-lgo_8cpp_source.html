14ImD createTestImD(unsigned int width, unsigned int height, const std::vector<RGBQUAD>& pixel_values = {}) {
15 BITMAPFILEHEADER fh = {};
16 BITMAPINFOHEADER ih = {};
17
18 ih.biWidth = width;
19 ih.biHeight = height;
20 ih.biBitCount = 24; // Most tests will assume 24-bit
21 ih.biSize = sizeof(BITMAPINFOHEADER);
22 ih.biCompression = 0; // BI_RGB
23 ih.biPlanes = 1;
24 ih.biSizeImage = width * height * 3; // Approximate size, can change with alignment
25
26 fh.bfType = 0x4D42; // 'BM'
27 fh.bfOffBits = sizeof(BITMAPFILEHEADER) + sizeof(BITMAPINFOHEADER);
28
29 int row_stride = (width * 3 + 3) & ~3;
30 ih.biSizeImage = row_stride * height;
31 fh.bfSize = fh.bfOffBits + ih.biSizeImage;
32
33
34 RGBQUAD* pixels_arr = nullptr;
35 if (width > 0 && height > 0) {
36 pixels_arr = new RGBQUAD[static_cast<size_t>(width) * height];
37 if (pixel_values.size() == static_cast<size_t>(width) * height) {
38 std::copy(pixel_values.begin(), pixel_values.end(), pixels_arr);
39 } else {
40 // Fill with black if data is not provided or size does not match
41 for (size_t i = 0; i < static_cast<size_t>(width) * height; ++i) {
42 pixels_arr[i] = {0, 0, 0}; // B, G, R
43 }
44 }
45 }
46
47 // ImD constructor creates a deep copy of pixelData
48 ImD img(fh, ih, pixels_arr, (4 - (width * 3) % 4) % 4);
49
50 delete[] pixels_arr; // pixels_arr was copied into img
51
52 return img; // ImD copy or move constructor is used
53}
